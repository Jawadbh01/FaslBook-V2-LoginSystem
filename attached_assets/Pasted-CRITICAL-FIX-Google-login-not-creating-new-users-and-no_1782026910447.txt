Replace the entire AuthProvider.tsx with EXACTLY this code:

"use client";
import { useEffect, useState } from "react";
import { onAuthStateChanged } from "firebase/auth";
import { doc, getDoc, setDoc, serverTimestamp } from "firebase/firestore";
import { auth, db } from "@/lib/firebase/config";
import { useAuthStore } from "@/store/authStore";

const PUBLIC = ["/login", "/email", "/register", "/create-farm", "/role-select", "/join-farm", "/pending"];

export default function AuthProvider({ children }: { children: React.ReactNode }) {
  const { setUser, setOrganization, setRole, setLoading } = useAuthStore();
  const [ready, setReady] = useState(false);

  useEffect(() => {
    const unsub = onAuthStateChanged(auth, async (firebaseUser) => {
      const path = window.location.pathname;

      if (!firebaseUser) {
        setUser(null);
        setOrganization(null);
        setRole(null);
        setLoading(false);
        setReady(true);
        if (!PUBLIC.includes(path)) {
          window.location.replace("/login");
        }
        return;
      }

      try {
        setUser(firebaseUser);
        const userRef = doc(db, "users", firebaseUser.uid);
        const userSnap = await getDoc(userRef);

        if (!userSnap.exists()) {
          // Brand new user - never registered before
          // Create basic user document with no role
          await setDoc(userRef, {
            id: firebaseUser.uid,
            name: firebaseUser.displayName || "",
            email: firebaseUser.email || "",
            phone: firebaseUser.phoneNumber || "",
            photoUrl: firebaseUser.photoURL || "",
            role: null,
            organizationId: null,
            status: "pending",
            createdAt: serverTimestamp(),
            updatedAt: serverTimestamp(),
            syncStatus: "synced",
          });
          setLoading(false);
          setReady(true);
          window.location.replace("/role-select");
          return;
        }

        // User document exists
        const userData = userSnap.data();
        setRole(userData.role);

        // No role selected yet - send to role select
        if (!userData.role) {
          setLoading(false);
          setReady(true);
          window.location.replace("/role-select");
          return;
        }

        // Has role but no organization
        if (!userData.organizationId) {
          setLoading(false);
          setReady(true);
          if (userData.role === "landlord") {
            window.location.replace("/create-farm");
          } else {
            // Check if has pending join request
            window.location.replace("/join-farm");
          }
          return;
        }

        // Has role and organization - fetch org and go to dashboard
        const orgSnap = await getDoc(doc(db, "organizations", userData.organizationId));
        if (orgSnap.exists()) {
          setOrganization(orgSnap.data() as any);
        }

        setLoading(false);
        setReady(true);

        // Only redirect if on a public page
        if (PUBLIC.includes(path)) {
          window.location.replace("/overview");
        }

      } catch (error) {
        console.error("Auth error:", error);
        setLoading(false);
        setReady(true);
      }
    });

    return () => unsub();
  }, []);

  if (!ready) {
    return (
      <div style={{
        minHeight: "100vh",
        display: "flex",
        flexDirection: "column",
        alignItems: "center",
        justifyContent: "center",
        backgroundColor: "white",
        gap: "16px"
      }}>
        <div style={{
          width: "48px",
          height: "48px",
          borderRadius: "50%",
          border: "4px solid #e5e7eb",
          borderTopColor: "#1B5E20",
          animation: "spin 0.8s linear infinite"
        }} />
        <p style={{ color: "#9ca3af", fontSize: "14px" }}>
          Loading FaslBook...
        </p>
        <style>{`@keyframes spin { to { transform: rotate(360deg); } }`}</style>
      </div>
    );
  }

  return <>{children}</>;
}
