Replace entire handleUserAfterAuth function with:

async function handleUserAfterAuth(uid: string, displayName: string | null, email: string | null, photoURL: string | null) {
  const userRef = doc(db, "users", uid);
  const userSnap = await getDoc(userRef);

  if (!userSnap.exists()) {
    await setDoc(userRef, {
      id: uid,
      name: displayName || "",
      email: email || "",
      phone: "",
      photoUrl: photoURL || "",
      role: null,
      organizationId: null,
      status: "pending",
      createdAt: serverTimestamp(),
      updatedAt: serverTimestamp(),
      syncStatus: "synced",
    });
    window.location.replace("/role-select");
    return;
  }

  const userData = userSnap.data();
  if (!userData.role) {
    window.location.replace("/role-select");
    return;
  }
  if (!userData.organizationId) {
    window.location.replace(userData.role === "landlord" ? "/create-farm" : "/join-farm");
    return;
  }
  window.location.replace("/overview");
}
