FEATURE 1 - Push Notifications using Firebase Cloud Messaging:

Install firebase messaging. In lib/firebase/config.ts add:
import { getMessaging, getToken, onMessage } from "firebase/messaging";

export const messaging = typeof window !== "undefined" ? getMessaging(app) : null;

export async function requestNotificationPermission(organizationId: string) {
  try {
    const permission = await Notification.requestPermission();
    if (permission !== "granted") return null;
    const { messaging } = await import("@/lib/firebase/config");
    if (!messaging) return null;
    const token = await getToken(messaging, {
      vapidKey: process.env.NEXT_PUBLIC_FIREBASE_VAPID_KEY,
    });
    if (token) {
      const { doc, setDoc } = await import("firebase/firestore");
      const { db, auth } = await import("@/lib/firebase/config");
      const user = auth.currentUser;
      if (user) {
        await setDoc(doc(db, "fcmTokens", user.uid), {
          token,
          organizationId,
          userId: user.uid,
          updatedAt: new Date(),
        });
      }
    }
    return token;
  } catch (err) {
    console.error("Notification permission error:", err);
    return null;
  }
}

FEATURE 2 - Complete PWA with Service Worker for offline:

Create public/sw.js - main service worker:
const CACHE_NAME = "faslbook-v2";
const STATIC_ASSETS = [
  "/",
  "/login",
  "/overview",
  "/manifest.json",
  "/icon-192.png",
  "/icon-512.png",
  "/logo.png",
];

self.addEventListener("install", (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

self.addEventListener("activate", (event) => {
  event.waitUntil(
    caches.keys().then((keys) =>
      Promise.all(keys.filter((k) => k !== CACHE_NAME).map((k) => caches.delete(k)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (event) => {
  if (event.request.method !== "GET") return;
  if (event.request.url.includes("firestore.googleapis.com")) return;
  if (event.request.url.includes("identitytoolkit.googleapis.com")) return;

  event.respondWith(
    caches.match(event.request).then((cached) => {
      if (cached) return cached;
      return fetch(event.request).then((response) => {
        if (response.ok && response.type === "basic") {
          const clone = response.clone();
          caches.open(CACHE_NAME).then((cache) => cache.put(event.request, clone));
        }
        return response;
      }).catch(() => caches.match("/") || new Response("Offline"));
    })
  );
});

# FaslBook Images Guide

## How to replace images:

1. **App Logo** (`public/logo.png`)
   - Size: 512x512 px
   - Format: PNG with transparent background
   - Used: Login screen, PWA icon

2. **PWA Icon Small** (`public/icon-192.png`)
   - Size: 192x192 px
   - Format: PNG
   - Used: Mobile home screen icon (small)

3. **PWA Icon Large** (`public/icon-512.png`)
   - Size: 512x512 px
   - Format: PNG
   - Used: Mobile home screen icon (large), splash screen

4. **Login Banner** (`public/banner.png`)
   - Size: 400x300 px
   - Format: PNG or JPG
   - Used: Login page illustration

5. **Splash Screen** (`public/splash.png`)
   - Size: 390x844 px (iPhone size)
   - Format: PNG
   - Used: PWA splash screen

## Steps:
1. Replace the files in /public folder with your images
2. Keep the same file names
3. Rebuild the app: npm run build
4. Deploy to Vercel

Also create simple green placeholder PNG files for icon-192.png and icon-512.png using this script.
Create scripts/create-icons.js:
const { createCanvas } = require("canvas");
const fs = require("fs");

function createIcon(size, filename) {
  const canvas = createCanvas(size, size);
  const ctx = canvas.getContext("2d");
  
  // Green background
  ctx.fillStyle = "#1B5E20";
  ctx.fillRect(0, 0, size, size);
  
  // White text
  ctx.fillStyle = "white";
  ctx.font = `bold ${size * 0.3}px Arial`;
  ctx.textAlign = "center";
  ctx.textBaseline = "middle";
  ctx.fillText("FB", size / 2, size / 2);
  
  fs.writeFileSync(`public/${filename}`, canvas.toBuffer("image/png"));
  console.log(`Created ${filename}`);
}

createIcon(192, "icon-192.png");
createIcon(512, "icon-512.png");
