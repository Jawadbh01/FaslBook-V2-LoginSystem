PART 2 - EXPORT FUNCTIONS

Create lib/exports/pdfExport.ts:

Install and use jsPDF and jspdf-autotable.

export async function exportToPDF(title: string, data: any[], columns: string[]) {
  const { jsPDF } = await import("jspdf");
  const autoTable = (await import("jspdf-autotable")).default;
  
  const doc = new jsPDF();
  
  // Header
  doc.setFillColor(27, 94, 32);
  doc.rect(0, 0, 210, 25, "F");
  doc.setTextColor(255, 255, 255);
  doc.setFontSize(16);
  doc.text("FaslBook - " + title, 14, 16);
  
  // Date
  doc.setFontSize(10);
  doc.text(new Date().toLocaleDateString("en-PK"), 160, 16);
  
  // Table
  autoTable(doc, {
    startY: 30,
    head: [columns],
    body: data,
    theme: "striped",
    headStyles: { fillColor: [27, 94, 32] },
  });
  
  doc.save(`FaslBook-${title}-${Date.now()}.pdf`);
}

Create lib/exports/excelExport.ts:

Use SheetJS (xlsx package).

export async function exportToExcel(title: string, data: any[], columns: string[]) {
  const XLSX = await import("xlsx");
  const ws = XLSX.utils.aoa_to_sheet([columns, ...data]);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, title);
  XLSX.writeFile(wb, `FaslBook-${title}-${Date.now()}.xlsx`);
}

"use client";
import { useEffect, useState } from "react";
import { Wifi, WifiOff } from "lucide-react";

export default function SyncIndicator() {
  const [online, setOnline] = useState(true);

  useEffect(() => {
    const handleOnline = () => setOnline(true);
    const handleOffline = () => setOnline(false);
    window.addEventListener("online", handleOnline);
    window.addEventListener("offline", handleOffline);
    setOnline(navigator.onLine);
    return () => {
      window.removeEventListener("online", handleOnline);
      window.removeEventListener("offline", handleOffline);
    };
  }, []);

  if (online) return null;

  return (
    <div
      className="fixed top-0 left-0 right-0 z-50 py-2 px-4 flex items-center justify-center gap-2 text-white text-sm font-medium"
      style={{ backgroundColor: "#E65100" }}
    >
      <WifiOff size={16} />
      <span>Offline - Changes will sync when connected</span>
    </div>
  );
}

Create lib/notifications/createNotification.ts:

export async function createNotification(params: {
  organizationId: string;
  title: string;
  description: string;
  type: "farm" | "inventory" | "finance" | "team";
  targetUserId?: string;
}) {
  const { addDoc, collection, serverTimestamp } = await import("firebase/firestore");
  const { db } = await import("@/lib/firebase/config");
  
  await addDoc(collection(db, "notifications"), {
    organizationId: params.organizationId,
    title: params.title,
    description: params.description,
    type: params.type,
    targetUserId: params.targetUserId || null,
    read: false,
    createdAt: serverTimestamp(),
    syncStatus: "synced",
  });
}

PART 5 - FIRESTORE SECURITY RULES

Create firestore.rules file in project root:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function getUserData() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }
    
    function isLandlord() {
      return isAuthenticated() && getUserData().role == "landlord";
    }
    
    function isManager() {
      return isAuthenticated() && getUserData().role == "manager";
    }
    
    function isFarmer() {
      return isAuthenticated() && getUserData().role == "farmer";
    }
    
    function isWorker() {
      return isAuthenticated() && getUserData().role == "worker";
    }
    
    function belongsToOrg(orgId) {
      return isAuthenticated() && getUserData().organizationId == orgId;
    }
    
    function canManage() {
      return isLandlord() || isManager();
    }
    
    // Users collection
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.auth.uid == userId;
      allow update: if isAuthenticated() && request.auth.uid == userId;
    }
    
    // Organizations
    match /organizations/{orgId} {
      allow read: if belongsToOrg(orgId);
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && belongsToOrg(orgId) && isLandlord();
    }
    
    // Join requests
    match /joinRequests/{requestId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if isAuthenticated();
    }
    
    // Parcels
    match /parcels/{parcelId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Crops
    match /crops/{cropId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Inventory
    match /inventoryItems/{itemId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    match /inventoryTransactions/{txId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
    }
    
    // Expenses
    match /expenses/{expId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Income
    match /income/{incId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
    }
    
    // Ledger - read only for users, system writes
    match /ledgerEntries/{ledgerId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
    }
    
    // Dealers
    match /dealers/{dealerId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    match /dealerTransactions/{txId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
    }
    
    // Loans
    match /loans/{loanId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Workers
    match /workers/{workerId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Attendance
    match /attendance/{attId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
      allow update: if isAuthenticated() && canManage();
    }
    
    // Worker payments
    match /workerPayments/{payId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated() && canManage();
    }
    
    // Notifications
    match /notifications/{notifId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated();
      allow update: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
    }
    
    // Activity logs
    match /activityLogs/{logId} {
      allow read: if isAuthenticated() && belongsToOrg(resource.data.organizationId);
      allow create: if isAuthenticated();
    }
  }
}
